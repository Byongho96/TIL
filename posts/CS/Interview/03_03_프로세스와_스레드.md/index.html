                                        int main() {
    int pipefd[2];
    char buffer[1024];
    pid_t pid;

    //  
    if (pipe(pipefd) == -1) {
        perror("pipe");
        exit(EXIT_FAILURE);
    }

    //   
    pid = fork();

    if (pid > 0) {
        //  
        close(pipefd[0]);  //   

        char message[] = "Hello, child process!";

        //   
        write(pipefd[1], message, strlen(message));

        //    
        close(pipefd[1]);

        printf("Parent process sent message to child process: %s\n", message);
    } else if (pid == 0) {
        //  
        close(pipefd[1]);  //   

        //   
        read(pipefd[0], buffer, sizeof(buffer));

        //    
        close(pipefd[0]);

        printf("Child process received message from parent process: %s\n", buffer);
    } else {
        perror("fork");
        exit(EXIT_FAILURE);
    }

    return 0;
}

                                        int main() {
  int fd;
  char buffer[BUFFER_SIZE];

  // Named Pipe 
  mkfifo("my_pipe", 0666);

  //   
  pid_t pid = fork();

  if (pid == 0) {
      //   ( )
      fd = open("my_pipe", O_RDONLY);
      read(fd, buffer, BUFFER_SIZE);
      printf(" :   - %s\n", buffer);
      close(fd);
  } else {
      //   ( )
      fd = open("my_pipe", O_WRONLY);
      write(fd, "!", 14);
      printf(" :  \n");
      close(fd);
  }

  return 0;
}
